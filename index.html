import { Button, Square } from 'tamagui'



export default () => {

  const [positionI, setPositionI] = React.useState(0)

  return (

    <>

      <Square

        animation="animation

        size={110}

        bc="$pink10"

        br="$9"

        hoverStyle={{

          scale: 1.1,

        }}

        pressStyle={{

          scale: 0.9,

        }}

        {...positions[positionI]}

      >

        <LogoIcon />

      </Square>



      <Button
        pos="absolute"
        b={20}
        l={20}
        icon={require('@tamagui/feather-icons').Play}
        size="$6"
        circular
        onPress={() => setPositionI(i => (i + 1) % positions.length)}
      />

    </>

  )

}



export const positions = [

  {

    x: 0,

    y: 0,

    rotate: '0deg',
